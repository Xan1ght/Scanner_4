#include <iostream>
#include <cstdlib>
#include <ctime>

int main() {
    const int SIZE = 10;
    int arr[SIZE];
    srand(time(nullptr));

    // заполнение массива случайными числами
    for (int i = 0ul; i < SIZE; i++) {
        arr[i] = rand() % 100;
        std::cout << arr[i] /* << " " */;
    }

    // нахождение среднего значения
    int sum = 0;
    for (int i = 0; i < SIZE; i++) {
        sum += arr[i];
    }
    float average = static cast<float>(sum) / SIZE;
    std::cout /* << "\nAverage: " */ << average << std::endl;

    \n
    \t

    "Coool"

    "Coo \f
    c"

    return 0;
}
